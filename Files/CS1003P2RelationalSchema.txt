CREATE TABLE IF NOT EXISTS `Movies`(`MovieID` INTEGER PRIMARY KEY AUTOINCREMENT, `Title` VARCHAR(30) NOT NULL, `Plot` VARCHAR(700) NOT NULL, `ReleaseDate` VARCHAR(10) NOT NULL, `RunTime` VARCHAR(6) NOT NULL, `Ratings` FLOAT(1));
CREATE TABLE IF NOT EXISTS `Actors`(`ActorID` INTEGER PRIMARY KEY AUTOINCREMENT, `FullName` VARCHAR(30) NOT NULL, `Birthday` VARCHAR(5));
CREATE TABLE IF NOT EXISTS `Awards`(`AwardID` INTEGER PRIMARY KEY AUTOINCREMENT, `AwardName` VARCHAR(100) NOT NULL, `MovieID` INTEGER, `ActorID` INTEGER, FOREIGN KEY (ActorID) REFERENCES Actors(`ActorID`), FOREIGN KEY (MovieID) REFERENCES Movies(`MovieID`));
CREATE TABLE IF NOT EXISTS `FilmCast`(`CastID` INTEGER PRIMARY KEY AUTOINCREMENT, `CharacterPlayed` VARCHAR(30) NOT NULL, `MovieID` INTEGER, `ActorID` INTEGER, FOREIGN KEY (ActorID) REFERENCES Actors(`ActorID`), FOREIGN KEY (MovieID) REFERENCES Movies(`MovieID`));
CREATE TABLE IF NOT EXISTS `FilmGenres`(`Genre` VARCHAR(20) NOT NULL,`MovieID` INTEGER NOT NULL, PRIMARY KEY (`Genre`, `MovieID`), FOREIGN KEY (MovieID) REFERENCES Movies(`MovieID`));
CREATE TABLE IF NOT EXISTS `FilmDirectors`(`Director` VARCHAR(20) NOT NULL,`MovieID` INTEGER NOT NULL, PRIMARY KEY (`Director`, `MovieID`), FOREIGN KEY (MovieID) REFERENCES Movies(`MovieID`));
